FNG_kill_QIE = {
	allowed = {
		original_tag = FNG
	}
	enable = {
		is_neighbor_of = QIE
		NOT = { is_in_faction_with = QIE }
		OR = {
			has_war_with = QIE
			has_completed_focus = FNG_Begin_the_War_of_National_Reclamation
			focus_progress = {
				focus = FNG_Begin_the_War_of_National_Reclamation
				progress > 0
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = conquer
		id = QIE
		value = 400
	}
	ai_strategy = {
		type = front_unit_request
		tag = QIE
		value = 100
	}
}

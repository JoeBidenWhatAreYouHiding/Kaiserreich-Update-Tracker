QIE_aggressive_push_with_buff = {
	allowed = { tag = QIE }
	enable = {
		has_idea = QIE_attack_FNG_idea
		has_war_with = FNG
		OR = {
			AND = {
				set_temp_variable = { QIE_aggro_day = global.num_days }
				modulo_temp_variable = { QIE_aggro_day = 30 }
				check_variable = { QIE_aggro_day < 7 }
			}
			AND = {
				716 = { is_controlled_by = QIE }
				FNG = {
					surrender_progress > 0.5
					has_capitulated = no
				}
				AND = {
					set_temp_variable = { QIE_aggro_day_2 = global.num_days }
					modulo_temp_variable = { QIE_aggro_day_2 = 30 }
					check_variable = { QIE_aggro_day_2 < 14 }
				}
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control

		country_trigger = {
			is_ally_with = FNG
		}

		state_trigger = {
			is_core_of = FNG
		}

		ratio = 0.1

		priority = 100
		ordertype = front
		execution_type = rush_weak
		execute_order = yes
	}
}
QIE_avoid_shanxi = {
	allowed = {
		original_tag = QIE
	}
	enable = {
		has_war_with = SHX
		NOT = { 799 = { is_controlled_by = SHX } }
		NOT = { 1072 = { is_controlled_by = SHX } }
		OR = {
			date > 1939.1.1
			AND = {
				set_temp_variable = { QIE_shanxi_day = global.num_days }
				modulo_temp_variable = { QIE_shanxi_day = 365 }
				OR = {
					check_variable = { QIE_shanxi_day > 300 }
					check_variable = { QIE_shanxi_day < 90 }
				}
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		tag = SHX
		ratio = 0.7
		priority = 100
		ordertype = front
		execution_type = careful
		execute_order = no
	}
}
QIE_play_safe_against_fengtian = {
	allowed = {
		original_tag = QIE
	}
	enable = {
		date < 1940.3.10
		is_in_faction = no
		has_war_with = FNG
		NOT = { has_idea = QIE_attack_FNG_idea }
		OR = {
			has_imminent_war = yes
			check_variable = { ROOT.potential_and_current_enemies^num > 1 } # Has more potential enemies than just Fengtian
			NOT = { surrender_progress > 0.02 }				#if at risk of being pushed, push Fengtian to start SSJW
		}
		NOT = {
			OR = {
				FNG = { surrender_progress > 0.5 }
				AND = {
					716 = { is_controlled_by = ROOT }
					FNG = { has_capitulated = no }
				}
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		tag = FNG
		ratio = 0.1
		priority = 700
		ordertype = front
		execution_type = careful
		execute_order = no
	}

	ai_strategy = {
		type = force_concentration_front_factor
		tag = FNG
		value = -100
	}
}

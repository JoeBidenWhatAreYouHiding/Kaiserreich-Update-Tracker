china_hates_japan = {
	allowed = {
		is_chinese_or_potential_chinese_tag = yes
	}
	enable = {
		country_exists = JAP
		JAP = { is_subject = no }
		if = {
			limit = {
				original_tag = KUM
			}
			has_country_flag = KUM_is_chinese_tag
		}
		is_working_with_JAP = no
		KOR = { any_core_state = { is_owned_by = JAP } }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = contain
		id = JAP
		value = 1000
	}
	ai_strategy = {
		type = antagonize
		id = JAP
		value = 1000
	}
	ai_strategy = {
		type = diplo_action_acceptance
		target = market_access_rights
		id = JAP
		value = -1000
	}
}

		NOT = { has_idea = CUF_call_of_the_united_front_idea_3 }

			is_controlled_by = CUF
		}

	ai_strategy = {
		type = garrison
		value = 50
	}

	ai_strategy = {
		type = front_unit_request
		country_trigger = {
			has_war_with = CUF
		}
		state_trigger = {
			NOT = { is_core_of = FROM.FROM }
		}
		value = -25
	}
}

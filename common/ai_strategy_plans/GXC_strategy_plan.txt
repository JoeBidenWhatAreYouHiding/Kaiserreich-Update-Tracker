			num_subjects > 0
			is_subject = yes

			surrender_progress > 0.4

GXC_block_war_with_beijing = {
	name = "GXC_starting_plan"
	desc = ""

	allowed = {
		original_tag = GXC
	}
	enable = {
		NOT = { has_active_mission = CHN_pressure_to_act_mission }
		NOT = { has_completed_focus = GXC_avenge_the_northern_expedition }
		NOT = { has_completed_focus = UPC_topple_beiyang }
		NOT = { has_war_with = QIE }
		NOT = { is_ally_with = QIE }
		QIE = { exists = yes }
		OR = {
			QIE = { has_war = no }
			has_war = yes
			AND = {
				613 = {
					owner = {
						is_ally_with = QIE
						strength_ratio = {
							tag = GXC
							ratio > 1
						}
					}
				}
				QIE = {
					surrender_progress < 0.20
					check_variable = { enemies^num < 2 }
				}
			}
		}
	}
	abort_when_not_enabled = yes

	focus_factors = {
		GXC_avenge_the_northern_expedition = 0.0
		UPC_topple_beiyang = 0.0
	}

		factor = 1.0
	}

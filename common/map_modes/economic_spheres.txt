scripted_map_modes = {
	economic_spheres_map_mode = {
		top = {
			type = none
		}
		bottom = {
			type = country
			color = {
				FROM = {
					if = {
						limit = { is_in_economic_sphere = no }
						set_temp_variable = { red = 0.7 }
						set_temp_variable = { blue = 0.7 }
						set_temp_variable = { green = 0.7 }
						set_temp_variable = { alpha = 0.7 }
					}
					else = {
						var:economic_sphere = {
							set_temp_variable = { red = THIS.red }
							multiply_temp_variable = { red = 0.7 }
							set_temp_variable = { green = THIS.green }
							multiply_temp_variable = { green = 0.7 }
							set_temp_variable = { blue = THIS.blue }
							multiply_temp_variable = { blue = 0.7 }
						}
						set_temp_variable = { alpha = 1 }
					}
				}
			}
			thickness = yes
		}

		far_text = country
		near_text = country

		update_daily = yes
	}
}

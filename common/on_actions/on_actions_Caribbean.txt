on_actions = {
	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
				limit = { FROM = { original_tag = WIF } }
				WIF_remove_CAN_ideas = yes
				WIF_reset_guyana_aluminium = yes
			}
		}
	}

	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					original_tag = DOM
					FROM = { original_tag = HAI }
				}
				country_event = { id = dominican_republic_events.89 days = 2 }
			}
			else_if = {
				limit = {
					original_tag = HAI
					FROM = { original_tag = DOM }
				}
				country_event = { id = dominican_republic_events.88 days = 2 }
			}
		}
	}

	#ROOT = attacking side
	#FROM = defending side
	#fired when two countries end up at war with each other (on_war is fired when a country goes to war against anyone and is not fired again when it enters war against another country unless it went to peace first)
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = {
						tag = CAN
						original_tag = WIF
					}
					OR = {
						FROM = { tag = CAN }
						FROM = { original_tag = WIF }
					}
				}
				WIF_remove_CAN_ideas = yes
				WIF_reset_guyana_aluminium = yes
			}
		}
	}
}

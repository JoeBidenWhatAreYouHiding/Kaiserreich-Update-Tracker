# Scripted AI desire for peace actions.
# See documentation.info file for more detailed scripting information.

peace_ai_desires = {

	# So GER gives it to BAT instead of taking it themselves
	treaty_of_moscow_GER_give_latgale_to_BAT = {
		peace_action_type = take_states
		enable = {
			NOT = { tag = ROOT.FROM } #Not Country is taker country
			is_ally_with = GER
			is_ally_with = BAT
			ROOT.FROM.FROM.FROM = { state = 880 } # Latgale (State to be taken by taker in peace conference)
			ROOT.FROM = { original_tag = BAT } #BAT will be taker country
		}
		ai_desire = 2000 #Massively offsets default take land yourself
	}

	# Case where MON involved in treaty of moscow but cannot take the land themselves (due to too low warscore)
	# AND Reichspakt cannot reach the land
	# Tells Reichspakt AI to give pan mongol lands to MON if unreachable
	treaty_of_moscow_reichspakt_give_pan_mongol_claims = {
		peace_action_type = take_states
		enable = {
			NOT = { tag = ROOT.FROM } #Not Country is taker country
			FROM.FROM = { original_tag = RUS } #Giver country is Russia
			is_ally_with = GER
			pc_is_on_same_side_as = MON
			NOT = { has_war_with = MON }
			ROOT.FROM = { original_tag = MON } #MON will be taker country
			ROOT.FROM.FROM.FROM = { #State to be taken by taker in peace conference
				OR = {
					is_core_of = MON
					is_claimed_by = MON
					has_variable = ambition_area_MON
				}
			}
			NOT = { #Do not give away any land if can reach MON
				is_neighbor_of = MON
				MON = { any_home_area_neighbor_country = { is_ally_with = GER } }
			}
		}
		ai_desire = 415 #Offsets do_not_grant_states_randomly and high enough to not take directly due to base 10 desire to take
		# base 415 - 400 (do_not_grant_states_randomly) = 15
		# 15 > 10 (base take states desire)
		# Low enough that it only beats give_states_other in take_states.txt
	}
}

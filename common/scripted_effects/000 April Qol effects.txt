	set_country_flag = game_will_crash

		limit = { has_country_flag = game_will_crash }
		crash_the_game = yes
	}
}

	add_to_temp_array = { transylvanian_states_array = 154.id } #Alfold
	add_to_temp_array = { transylvanian_states_array = 76.id } #Cluj
	add_to_temp_array = { transylvanian_states_array = 84.id } #Brasov
	add_to_temp_array = { transylvanian_states_array = 789.id } #Szeklerland
	hidden_effect = {
		for_each_scope_loop = {
			array = transylvanian_states_array
			random_list = {
				seed = global.date
				50 = {
					set_temp_variable = { tag_to_release = event_target:central_hungary }
				}
				50 = {
					set_temp_variable = { tag_to_release = event_target:muntenia }
				}
			}
			var:v = { transfer_if_owned = yes }
		}
	}
}

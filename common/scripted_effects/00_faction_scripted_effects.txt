@num_faction_attributes = 7
save_former_faction_attributes = {
	resize_array = { former_faction_attributes = @num_faction_attributes }
	# former_faction_attributes^0 = faction_research
	# former_faction_attributes^1 = faction_military
	# former_faction_attributes^2 = spymaster
	# former_faction_attributes^3 = land_facility_assigned
	# former_faction_attributes^4 = naval_facility_assigned
	# former_faction_attributes^5 = air_facility_assigned
	# former_faction_attributes^6 = nuclear_facility_assigned
	if = {
		limit = { has_faction_research_unlocked = yes }
		set_variable = { former_faction_attributes^0 = 1 }
	}
	if = {
		limit = { has_faction_military_unlocked = yes }
		set_variable = { former_faction_attributes^1 = 1 }
	}
	if = {
		limit = {
			has_dlc_lar = yes
			any_of_scopes = {
				array = faction_members
				is_spymaster = yes
			}
		}
		set_variable = { former_faction_attributes^2 = 1 }
	}
	# Note: below checks will sometimes be inaccurate but it is the best prediction available of whether facility progress has been undertaken.
	# For reference, these hold true when there is a joint facility assigned and said facility has available supportive scientist slots.
	if = {
		limit = {
			any_of_scopes = {
				array = faction_members
				can_assign_supportive_scientist_to_faction = specialization_land
			}
		}
		set_variable = { former_faction_attributes^3 = 1 }
	}
	if = {
		limit = {
			any_of_scopes = {
				array = faction_members
				can_assign_supportive_scientist_to_faction = specialization_naval
			}
		}
		set_variable = { former_faction_attributes^4 = 1 }
	}
	if = {
		limit = {
			any_of_scopes = {
				array = faction_members
				can_assign_supportive_scientist_to_faction = specialization_air
			}
		}
		set_variable = { former_faction_attributes^5 = 1 }
	}
	if = {
		limit = {
			any_of_scopes = {
				array = faction_members
				can_assign_supportive_scientist_to_faction = specialization_nuclear
			}
		}
		set_variable = { former_faction_attributes^6 = 1 }
	}
}

apply_former_faction_attributes = {
	if = {
		limit = { has_variable = former_faction_attributes^num }
		# former_faction_attributes^0 = faction_research
		# former_faction_attributes^1 = faction_military
		# former_faction_attributes^2 = spymaster
		# former_faction_attributes^3 = land_facility_assigned
		# former_faction_attributes^4 = naval_facility_assigned
		# former_faction_attributes^5 = air_facility_assigned
		# former_faction_attributes^6 = nuclear_facility_assigned
		if = {
			limit = { check_variable = { former_faction_attributes^0 = 1 } }
			set_faction_research_unlocked = yes
		}
		if = {
			limit = { check_variable = { former_faction_attributes^1 = 1 } }
			set_faction_military_unlocked = yes
		}
		if = {
			limit = { check_variable = { former_faction_attributes^2 = 1 } }
			set_faction_spymaster = yes
		}
		# Note: below checks will sometimes be inaccurate but it is the best prediction available of whether facility progress has been undertaken.
		# For reference, these hold true when there is a joint facility assigned and said facility has available supportive scientist slots.
		if = {
			limit = { check_variable = { former_faction_attributes^3 = 1 } }
			add_faction_initiative = 1
		}
		if = {
			limit = { check_variable = { former_faction_attributes^4 = 1 } }
			add_faction_initiative = 1
		}
		if = {
			limit = { check_variable = { former_faction_attributes^5 = 1 } }
			add_faction_initiative = 1
		}
		if = {
			limit = { check_variable = { former_faction_attributes^6 = 1 } }
			add_faction_initiative = 1
		}
	}
	clear_array = former_faction_attributes
}

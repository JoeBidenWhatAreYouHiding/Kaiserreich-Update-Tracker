### Transfer Technology
### Written by wyandotte
### Automated by Dr_Njitram
#
# How to use:
#
#	TAG = { transfer_technology = yes }
#
# As it uses 'PREV', you need to make sure the previous TAG is the one you transfer from
# And the TAG you transfer *to* is the scope in which the effect is called.
#
# For example, an event fires for a given country that wants to transfer techs to ENG
# In an event option:
#
# ENG = { transfer_technology = yes }
#
# If a third country should be the source of the techs:
# FRA = {
#	ENG = { transfer_technology = yes }
# }

		### Mastery Transfer - works for all tracks except special forces ###
		# Note: mastery can only be added if a subdoctrine was previously applied.
		if = {
			limit = { has_variable = doctrine_tracks^num }
			for_each_loop = {
				array = doctrine_tracks
				value = doctrine_track

				set_temp_variable = { mastery_check = 500 }
				set_temp_variable = { mastery_loops_num = { value = mastery_check multiply = 0.1 } }
				for_loop_effect = {
					end = mastery_loops_num
					break = mastery_determined
					if = {
						limit = {
							PREV = {
								meta_trigger = {
									text = { has_mastery = { amount = [MASTERY] track = [TRACK] } }
									MASTERY = "[?mastery_check|0]"
									TRACK = "[?doctrine_track.GetTokenKey]"
								}
							}
						}
						meta_effect = {
							text = { add_mastery = { amount = [MASTERY] track = [TRACK] } }
							MASTERY = "[?mastery_check|0]"
							TRACK = "[?doctrine_track.GetTokenKey]"
						}
						set_temp_variable = { mastery_determined = 1 }
					}
					else = {
						subtract_from_temp_variable = { mastery_check = 10 }
					}
				}
			}

					set_sub_doctrine = { sub_doctrine = rangers_2 track = 0 }
				}
			}
			### Mastery Transfer Special Forces ###
			# Less precise than the other subdoctrines as no way to differentiate shared subdoctrines in different tracks
			if = {
				limit = { has_variable = special_forces_doctrines^num }
				for_each_loop = {
					array = special_forces_doctrines
					value = special_forces_doctrine

					set_temp_variable = { mastery_levels_check = 5 }
					set_temp_variable = { mastery_levels_loops = 5 }
					for_loop_effect = {
						end = mastery_levels_loops
						break = mastery_determined
						if = {
							limit = {
								PREV = {
									meta_trigger = {
										text = { has_mastery_level = { amount = [MASTERY] sub_doctrine = [SUBDOCTRINE] } }
										MASTERY = "[?mastery_levels_check|0]"
										SUBDOCTRINE = "[?special_forces_doctrine.GetTokenKey]"
									}
								}
							}
							set_temp_variable = { mastery_add = { value = mastery_levels_check multiply = 100 } }
							meta_effect = {
								text = { add_mastery = { amount = [MASTERY] sub_doctrine = [SUBDOCTRINE] } }
								MASTERY = "[?mastery_add|0]"
								SUBDOCTRINE = "[?special_forces_doctrine.GetTokenKey]"
							}
							set_temp_variable = { mastery_determined = 1 }
						}
						else = {
							subtract_from_temp_variable = { mastery_levels_check = 1 }
						}
					}
				}
			}
		}
	}
}

transfer_technologies = {
	#inherit_technology = PREV #DO NOT USE THIS
	for_each_loop = {
		array = PREV.researched_techs
		value = technology
		if = {
			limit = {
				# the rest are not mutually exclusive
				NOT = { has_tech = var:technology }
				# Account for industry techs
				if = {
					limit = { check_variable = { technology = token:flexible_line } }
					NOT = { has_tech = streamlined_line }
				}
				else_if = {
					limit = { check_variable = { technology = token:streamlined_line } }
					NOT = { has_tech = flexible_line }
				}
				else_if = {
					limit = {
						OR = {
							check_variable = { technology = token:concentrated_industry }
							check_variable = { technology = token:concentrated_industry2 }
							check_variable = { technology = token:concentrated_industry3 }
							check_variable = { technology = token:concentrated_industry4 }
							check_variable = { technology = token:concentrated_industry5 }
						}
					}
					NOT = { has_tech = dispersed_industry }
				}
				else_if = {
					limit = {
						OR = {
							check_variable = { technology = token:dispersed_industry }
							check_variable = { technology = token:dispersed_industry2 }
							check_variable = { technology = token:dispersed_industry3 }
							check_variable = { technology = token:dispersed_industry4 }
							check_variable = { technology = token:dispersed_industry5 }
						}
					}
					NOT = { has_tech = concentrated_industry }
				}
			}
			set_technology = { var:technology = 1 popup = no }
		}
	}
}

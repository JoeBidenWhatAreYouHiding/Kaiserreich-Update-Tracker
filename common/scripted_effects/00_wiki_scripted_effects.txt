### SCRIPTED EFFECTS FOR THE WIKI ###
### NEVER EXECUTE THESE DURING THE COURSE OF THE GAME ###
### FIRE VIA SOME PLACE LIKE AN EVENT TO GET THE DATA ###

gather_country_data_1936_start = {
	# Data is intended for webpage: https://kaiserreich.wiki/Countries
	# Country Names
	every_country = {
		log = "[THIS.GetName]"
	}
	log = "L ----------------------------------------"
	log = "TAG | Ideology | Owned States | Research Slots | Core Pop | Non Core Pop | Building Slots | Num Mils | Num Docks | Num Civs |"
	every_country = {
		set_temp_variable = { core_population = 0 }
		set_temp_variable = { non_core_population = 0 }
		set_temp_variable = { number_of_building_slots = 0 }
		every_owned_state = {
			if = {
				limit = { is_core_of = PREV }
				add_to_temp_variable = { core_population = state_population_k }
			}
			else = {
				add_to_temp_variable = { non_core_population = state_population_k }
			}
			add_to_temp_variable = { number_of_building_slots = modifier@local_building_slots }
		}
		divide_temp_variable = { core_population = 1000 } #To make population be in millions
		divide_temp_variable = { non_core_population = 1000 } #To make population be in millions
		log = "[THIS.GetTag] | [?current_party_ideology_group.GetTokenLocalizedKey] | [?num_owned_states] | [?original_research_slots] | [?core_population|2] | [?non_core_population|2] | [?number_of_building_slots] | [?num_of_military_factories] | [?num_of_naval_factories] | [?num_of_civilian_factories]"
	}
	log = "L ----------------------------------------"
	# Resources
	log = "TAG | Oil | Aluminium | Rubber | Tungsten | Steel | Chromium"
	every_country = {
		log = "[THIS.GetTag] | [?resource_produced@oil] | [?resource_produced@aluminium] | [?resource_produced@rubber] | [?resource_produced@tungsten] | [?resource_produced@steel] | [?resource_produced@chromium]"
	}
}

india_claim_ceylon_effect = {
	CEY = {
		effect_tooltip = {
			every_core_state = {
				limit = { NOT = { is_core_of = ROOT } }
				add_claim_by = ROOT
				enable_state_integration_80_tooltip = yes
			}
		}
		hidden_effect = {
			every_core_state = {
				add_claim_of_ROOT = yes
				ROOT = { PREV = { enable_state_integration_80_compliance = yes } }
			}
		}
	}
	if = {
		limit = {
			CEY = {
				any_core_state = {
					OR = {
						is_owned_by = GER
						owner = { is_subject_of = GER }
					}
					is_controlled_by = owner
				}
			}
			OR = {
				NOT = { GER = { has_opinion_modifier = CAN_halifax_conference } }
				NOT = { is_in_faction_with = ENT }
			}
		}
		if = {
			limit = {
				is_in_faction_with = GER
				NOT = { owns_state = 422 }
			}
			custom_effect_tooltip = tooltip_white_line
			GER = { country_event = india_events.61 }
			custom_effect_tooltip = if_they_accept_tt
			effect_tooltip = {
				every_state = {
					limit = {
						OR = {
							is_ceylon = yes
							state = 818 #Pondicherry
						}
						OR = {
							is_owned_by = GER
							owner = { is_subject_of = GER }
						}
						is_controlled_by = owner
					}
					transfer_state_to = ROOT
				}
			}
		}
		else_if = {
			limit = {
				GER = {
					has_war = yes
					any_enemy_country = { is_actual_major = yes }
				}
			}
			custom_effect_tooltip = tooltip_white_line
			GER = { country_event = india_events.60 }
			effect_tooltip = {
				every_state = {
					limit = {
						OR = {
							is_ceylon = yes
							state = 818 #Pondicherry
						}
						OR = {
							is_owned_by = GER
							owner = { is_subject_of = GER }
						}
						is_controlled_by = owner
					}
					transfer_state_to = ROOT
				}
			}
		}
		else_if = {
			limit = {
				CEY = {
					any_core_state = {
						owner = {
							OR = {
								tag = GER
								is_subject_of = GER
							}
						}
					}
				}
				OR = {
					NOT = { GER = { has_opinion_modifier = CAN_halifax_conference } }
					NOT = { is_in_faction_with = ENT }
				}
			}
			custom_effect_tooltip = tooltip_white_line
			custom_effect_tooltip = india_will_seize_ceylon_if_germany_at_war
		}
	}
	else_if = {
		limit = { has_subject = CEY }
		custom_effect_tooltip = tooltip_white_line
		annex_country = {
			target = CEY
			transfer_troops = no
		}
	}
	else_if = {
		limit = { can_declare_war_on = CEY }
		custom_effect_tooltip = tooltip_white_line
		create_wargoal = {
			type = annex_everything
			target = CEY
		}
	}
}

		}

			remove_idea = india_reconstruction_5

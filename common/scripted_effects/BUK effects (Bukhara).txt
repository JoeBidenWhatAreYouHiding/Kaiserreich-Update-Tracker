BUK_demilitarise_samarkand = {
	732 = { #Samarkand
		if = {
			limit = { is_owned_by = ROOT }
			set_demilitarized_zone = yes
		}
	}
}

BUK_spawn_ibrahim_bek_remnant_division = {
	if = {
		limit = { has_template = "Ibrahim Bek's Riders" }
		delete_unit_template_and_units = { division_template = "Ibrahim Bek's Riders" }
	}
	if = {
		limit = { NOT = { has_variable = BUK_cavalry_battalion_count } }
		set_temp_variable = { BUK_cavalry_battalion_count = 1 }
	}
	custom_override_tooltip = {
		tooltip = {
			localization_key = tooltip_create_division_template
			NAME = "Ibrahim Bek's Remnants"
			LINE_1 = { localization_key = line_battalion_tt NAME = cavalry ICON = GFX_unit_cavalry_icon_small COUNT = "[?BUK_cavalry_battalion_count]" }
		}
		if = {
			limit = { has_template = "Ibrahim Bek's Remnants" }
			delete_unit_template_and_units = { division_template = "Ibrahim Bek's Remnants" disband = yes } #no cheating
		}
		division_template = {
			name = "Ibrahim Bek's Remnants"
			division_names_group = CA_CAV_01
			regiments = {
				cavalry = { x = 0 y = 0 }
			}
		}
		if = {
			limit = { check_variable = { BUK_cavalry_battalion_count > 1 } }
			while_loop_effect = {
				limit = { check_variable = { BUK_cavalry_battalion_count > 1 } }
				add_units_to_division_template = {
					template_name = "Ibrahim Bek's Remnants"
					regiments = { cavalry = 0 }
				}
				add_to_temp_variable = { BUK_cavalry_battalion_count = -1 }
			}
		}
	}
	capital_scope = {
		custom_override_tooltip = {
			tooltip = {
				localization_key = tooltip_spawn_unit
				TYPE = "Ibrahim Bek's Remnants"
			}
			create_unit = {
				division = "division_template = \"Ibrahim Bek's Remnants\" start_experience_factor = 0.2"
				prioritize_location = 1573
				owner = ROOT
			}
		}
	}
}

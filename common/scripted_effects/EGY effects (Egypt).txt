EGY_create_mandist_militia_divisions = {
	log = "[GetLogRoot]: Create divisions EGY_create_mandist_militia_divisions"
	custom_override_tooltip = {
		tooltip = {
			localization_key = tooltip_create_division_template
			NAME = "Al-Milishiya"
			LINE_1 = { localization_key = line_battalion_tt NAME = irregular_infantry ICON = GFX_unit_irregular_infantry_icon_small COUNT = 4 }
		}
		if = {
			limit = { has_template = "Al-Milishiya" }
			delete_unit_template_and_units = { division_template = "Al-Milishiya" disband = yes } #no cheating
		}
		division_template = {
			name = "Al-Milishiya"
			division_names_group = EGY_MIL_01
			regiments = {
				irregular_infantry = { x = 0 y = 0 }
				irregular_infantry = { x = 0 y = 1 }
				irregular_infantry = { x = 1 y = 0 }
				irregular_infantry = { x = 1 y = 1 }
			}
			template_counter = 68
		}
	}
	if = {
		limit = { tag = SUD }
		capital_scope = {
			custom_override_tooltip = {
				tooltip = {
					localization_key = tooltip_spawn_units
					COUNT = 7
					TYPE = "Al-Milishiya"
				}
				create_unit = {
					division = "division_template = \"Al-Milishiya\" start_experience_factor = 0.05"
					owner = PREV
					count = 7
				}
			}
		}
		random_owned_controlled_state = {
			limit = { valid_state_for_spawning_friendly_unit = yes }
			custom_override_tooltip = {
				tooltip = {
					localization_key = tooltip_spawn_units
					COUNT = 3
					TYPE = "Al-Milishiya"
				}
				create_unit = {
					division = "division_template = \"Al-Milishiya\" start_experience_factor = 0.05"
					owner = PREV
					count = 3
				}
			}
		}
		random_owned_controlled_state = {
			limit = { valid_state_for_spawning_friendly_unit = yes }
			custom_override_tooltip = {
				tooltip = {
					localization_key = tooltip_spawn_units
					COUNT = 3
					TYPE = "Al-Milishiya"
				}
				create_unit = {
					division = "division_template = \"Al-Milishiya\" start_experience_factor = 0.05"
					owner = PREV
					count = 3
				}
			}
		}
	}
	else = {
		random_owned_controlled_state = {
			limit = { valid_state_for_spawning_friendly_unit = yes }
			prioritize = { 1010 }
			custom_override_tooltip = {
				tooltip = {
					localization_key = tooltip_spawn_units
					COUNT = 2
					TYPE = "Al-Milishiya"
				}
				create_unit = {
					division = "division_template = \"Al-Milishiya\" start_experience_factor = 0.05"
					owner = PREV
					count = 2
				}
			}
		}
	}
}

EGY_set_kingdom_cosmetic_tag = {
	if = {
		limit = {
			NOT = { has_cosmetic_tag = EGY_kingdom }
			NOT = { has_cosmetic_tag = EGY_khedivate }
		}
		set_cosmetic_tag = EGY_kingdom
	}
}

EGY_set_khedivate_cosmetic_tag = {
	if = {
		limit = { NOT = { has_cosmetic_tag = EGY_khedivate } }
		set_cosmetic_tag = EGY_khedivate
	}
}

EGY_remove_monarchy = {
	if = {
		limit = {
			OR = {
				has_cosmetic_tag = EGY_kingdom
				has_cosmetic_tag = EGY_khedivate
			}
		}
		drop_cosmetic_tag = yes
	}
	if = {
		limit = { has_character = EGY_fuad_i }
		EGY_fuad_i = {
			hidden_effect = { remove_all_country_leader_roles = yes }
			retire = yes
		}
	}
	if = {
		limit = { has_character = EGY_farouk_i }
		EGY_farouk_i = {
			hidden_effect = { remove_all_country_leader_roles = yes }
			retire = yes
		}
	}
	if = {
		limit = { has_character = EGY_abbas_ii }
		EGY_abbas_ii = {
			hidden_effect = { remove_all_country_leader_roles = yes }
			retire = yes
		}
	}
	if = {
		limit = { has_character = EGY_abdel_moneim_i }
		EGY_abdel_moneim_i = {
			hidden_effect = { remove_all_country_leader_roles = yes }
			retire = yes
		}
	}
}

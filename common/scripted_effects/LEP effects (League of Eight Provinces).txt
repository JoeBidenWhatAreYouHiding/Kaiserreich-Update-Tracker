
LEP_modify_tsingtau_spirit = {
	if = {
		limit = { has_variable = LEP_tsingtau_level }
		### Calculation ###
		# Get Old LEP_tsingtau_level for meta effect later
		set_temp_variable = { LEP_tsingtau_level_old = LEP_tsingtau_level }

		# Get new LEP party pop before effect for party pop executes
		set_temp_variable = { LEP_var_popularity = { value = party_popularity@market_liberal add = LEP_market_liberal_pp_change } }

		# Determine new LEP_tsingtau_level
		set_temp_variable = {
			LEP_tsingtau_level_new = {
				value = 0
				if = { limit = { value = LEP_var_popularity less_than = 0.05 } add = 0 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.15 } add = 1 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.20 } add = 2 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.24 } add = 3 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.28 } add = 4 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.31 } add = 5 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.34 } add = 6 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.37 } add = 7 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.40 } add = 8 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.42 } add = 9 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.44 } add = 10 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.46 } add = 11 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.48 } add = 12 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.50 } add = 13 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.60 } add = 14 }
				else_if = { limit = { value = LEP_var_popularity less_than = 0.75 } add = 15 }
				else = { add = 16 }
			}
		}

		### Tooltip ###
		if = {
			limit = { NOT = { check_variable = { LEP_market_liberal_pp_change = 0 } } }
			add_popularity = { ideology = market_liberal popularity = LEP_market_liberal_pp_change }
		}
		if = {
			limit = {
				NOT = { check_variable = { LEP_tsingtau_level_new = LEP_tsingtau_level_old } }
			}
			meta_effect = {
				text = {
					swap_ideas = {
						remove_idea = LEP_Tsingtau_Collaboration[TSINGTAU_LEVEL_OLD]
						add_idea = LEP_Tsingtau_Collaboration[TSINGTAU_LEVEL_NEW]
					}
				}
				TSINGTAU_LEVEL_OLD = "[?LEP_tsingtau_level_old]"
				TSINGTAU_LEVEL_NEW = "[?LEP_tsingtau_level_new]"
			}
		}
		else = {
			meta_effect = {
				text = {
					custom_effect_tooltip = {
						localization_key = LEP_Tsingtau_Collaboration_idea_unchanged
						IDEA = LEP_Tsingtau_Collaboration[TSINGTAU_LEVEL_OLD]
					}
				}
				TSINGTAU_LEVEL_OLD = "[?LEP_tsingtau_level_old]"
			}
		}

		### Additonal Hidden Effects ###
		hidden_effect = {
			set_variable = { LEP_tsingtau_level = LEP_tsingtau_level_new }
			if = {
				limit = { check_variable = { LEP_tsingtau_level = 3 } }
				country_event = league_german_influence_events.50
			}
			else_if = {
				limit = { check_variable = { LEP_tsingtau_level = 4 } }
				country_event = league_german_influence_events.53
			}
			else_if = {
				limit = { check_variable = { LEP_tsingtau_level = 5 } }
				country_event = league_german_influence_events.31
			}
			else_if = {
				limit = { check_variable = { LEP_tsingtau_level = 7 } }
				country_event = league_german_influence_events.54
			}
			else_if = {
				limit = { check_variable = { LEP_tsingtau_level = 8 } }
				country_event = league_german_influence_events.55
			}
			else_if = {
				limit = { check_variable = { LEP_tsingtau_level = 13 } }
				country_event = league_german_influence_events.56
			}
			else_if = {
				limit = { check_variable = { LEP_tsingtau_level = 14 } }
				country_event = league_german_influence_events.33
			}

			if = {
				limit = {
					check_variable = { LEP_tsingtau_level > 0 }
				}
				GEA = {
					# Refresh docking rights - there are no triggers to check docking rights
					# Applying it to the tag that already has docking rights causes error in log
					diplomatic_relation = {
						country = LEP
						relation = docking_rights
						active = no
					}
					diplomatic_relation = {
						country = LEP
						relation = docking_rights
						active = yes
					}
				}
				if = {
					limit = {
						check_variable = { LEP_tsingtau_level > 1 }
						NOT = { GEA = { has_military_access_to = LEP } }
					}
					GEA = {
						diplomatic_relation = {
							country = LEP
							relation = military_access
							active = yes
						}
					}
				}
			}
		}
	}
}

LEP_reduce_damaged_industry = {
	if = {
		limit = {
			has_idea = LEP_Damaged_Industry
		}
		swap_ideas = {
			remove_idea = LEP_Damaged_Industry
			add_idea = LEP_Damaged_Industry1
		}
	}
	else_if = {
		limit = {
			has_idea = LEP_Damaged_Industry1
		}
		swap_ideas = {
			remove_idea = LEP_Damaged_Industry1
			add_idea = LEP_Damaged_Industry2
		}
	}
	else_if = {
		limit = {
			has_idea = LEP_Damaged_Industry2
		}
		remove_ideas = LEP_Damaged_Industry2
	}
}

LEP_increase_kmt_unrest_level = {
	add_to_variable = { lep_kmt_unrest_level = 1 }
	clamp_variable = { var = lep_kmt_unrest_level min = 0 max = 4 }

	set_variable = {
		lep_suppress_kmt_mission_timeout = {
			value = 120
			subtract = { value = 20 multiply = lep_kmt_unrest_level }
		}
	}
}

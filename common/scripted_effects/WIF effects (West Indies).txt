WIF_evaluate_tensions = {
	hidden_effect = {
		WIF = {
			set_variable = { WIF_social_turmoil_stability = 0 }
			set_variable = { WIF_social_turmoil_war_support = 0 }
			set_variable = { WIF_social_turmoil_political_power = 0 }

			#Socialist baseline
			if = {
				limit = { has_socialist_government = no }
				if = {
					limit = { check_variable = { WIF_socialist_tensions_var > 80 } }
					subtract_from_variable = { WIF_social_turmoil_stability = 0.2 }
					subtract_from_variable = { WIF_social_turmoil_war_support = 0.15 }
					country_event = { id = west_indies_federation_politics_events.31 days = 2 random_days = 3 } # The Socialist Revolution
				}
				else_if = {
					limit = { check_variable = { WIF_socialist_tensions_var > 60 } }
					subtract_from_variable = { WIF_social_turmoil_stability = 0.15 }
					subtract_from_variable = { WIF_social_turmoil_war_support = 0.1 }
				}
				else_if = {
					limit = { check_variable = { WIF_socialist_tensions_var > 40 } }
					subtract_from_variable = { WIF_social_turmoil_stability = 0.1 }
					subtract_from_variable = { WIF_social_turmoil_war_support = 0.05 }
				}
				else_if = {
					limit = { check_variable = { WIF_socialist_tensions_var > 20 } }
					subtract_from_variable = { WIF_social_turmoil_stability = 0.05 }
				}
			}

			#Imperial baseline
			if = {
				limit = { WIF_patriotic_union_path = no }
				if = {
					limit = { check_variable = { WIF_empire_tensions_var > 80 } }
					subtract_from_variable = { WIF_social_turmoil_political_power = 0.2 }
					subtract_from_variable = { WIF_social_turmoil_war_support = 0.15 }
					country_event = { id = west_indies_federation_politics_events.32 days = 2 random_days = 3 } # The Governor-General's Coup
				}
				else_if = {
					limit = { check_variable = { WIF_empire_tensions_var > 60 } }
					subtract_from_variable = { WIF_social_turmoil_political_power = 0.15 }
					subtract_from_variable = { WIF_social_turmoil_war_support = 0.1 }
				}
				else_if = {
					limit = { check_variable = { WIF_empire_tensions_var > 40 } }
					subtract_from_variable = { WIF_social_turmoil_political_power = 0.1 }
					subtract_from_variable = { WIF_social_turmoil_war_support = 0.05 }
				}
				else_if = {
					limit = { check_variable = { WIF_empire_tensions_var > 20 } }
					subtract_from_variable = { WIF_social_turmoil_political_power = 0.05 }
				}
			}

			if = {
				limit = { WIF_patriotic_union_path = yes }
				multiply_variable = { WIF_social_turmoil_political_power = 0 }
				multiply_variable = { WIF_social_turmoil_stability = 1.5 } #not two in order to make up for initial hit after a soc/imp coup
			}
			else_if = {
				limit = { has_socialist_government = yes }
				multiply_variable = { WIF_social_turmoil_political_power = 1.5 } #not two in order to make up for initial hit after a soc/imp coup
				multiply_variable = { WIF_social_turmoil_stability = 0 }
			}
			force_update_dynamic_modifier = yes
		}
	}
}

		activate_targeted_decision = { target = THIS decision = WIF_integration_process_NCP_french_conscription }

	}
}

WIF_reset_guyana_aluminium = {
	# Removes the resource modifiers on Cayenne/Saguenay states before NCP mechanics are finished
	hidden_effect = {
		if = {
			limit = { 310 = { has_dynamic_modifier = { modifier = WIF_bauxite_mine_strike } } }
			310 = { remove_dynamic_modifier = { modifier = WIF_bauxite_mine_strike } }
		}
		if = {
			limit = { 310 = { has_state_flag = WIF_NCP_royal_assistance_aluminium } }
			310 = { clr_state_flag = WIF_NCP_royal_assistance_aluminium }
			add_resource = {
				type = aluminium
				amount = 50
				state = 310
			}
			add_resource = {
				type = aluminium
				amount = -10
				state = 466
			}
		}
	}
}

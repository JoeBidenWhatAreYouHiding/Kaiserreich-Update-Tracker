XSM_change_yushu_tensions = {
	add_to_variable = { XSM.XSM_yushu_tensions = XSM_yushu_tensions_change tooltip = XSM_yushu_tension_increase_tooltip }
	hidden_effect = {
		XSM = {
			if = {
				limit = {
					NOT = { check_variable = { XSM_yushu_tensions < 100 } }
					NOT = { has_war_with = TIB }
				}
				country_event = ma_clique_events.200
			}
		}
	}
}

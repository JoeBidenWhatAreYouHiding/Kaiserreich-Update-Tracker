scripted_gui = {
	kr_intro_screen_button = {
		context_type = player_context
		window_name = "kr_intro_screen_button_container"
		parent_window_token = top_bar
		dirty = kr_intro_screen_variable

		ai_enabled = { always = no }

		effects = {
			kr_intro_screen_button_click = {
				if = {
					limit = { has_variable = kr_intro_screen_variable }
					clear_variable = kr_intro_screen_variable
				}
				else = {
					set_variable = { kr_intro_screen_variable = 1 }
				}
				set_variable_to_random = kr_intro_screen_refresh
			}
		}
	}

	kr_intro_screen = {
		context_type = player_context
		window_name = "kr_intro_screen_container"
		dirty = kr_intro_screen_refresh

		ai_enabled = { always = no }

		visible = {
			has_variable = kr_intro_screen_variable
		}

		triggers = {
			tab_2_visible = { tag_has_path_guide = yes }

			tab_1_click_enabled = { NOT = { check_variable = { kr_intro_screen_variable = 1 } } }
			tab_2_click_enabled = { NOT = { check_variable = { kr_intro_screen_variable = 2 } } }
			tab_3_click_enabled = { NOT = { check_variable = { kr_intro_screen_variable = 3 } } }
			tab_4_click_enabled = { NOT = { check_variable = { kr_intro_screen_variable = 4 } } }
		}

		effects = {
			tab_1_click = {
				set_variable = { kr_intro_screen_variable = 1 }
				set_variable_to_random = kr_intro_screen_refresh
			}
			tab_2_click = {
				set_variable = { kr_intro_screen_variable = 2 }
				set_variable_to_random = kr_intro_screen_refresh
			}
			tab_3_click = {
				set_variable = { kr_intro_screen_variable = 3 }
				set_variable_to_random = kr_intro_screen_refresh
			}
			tab_4_click = {
				set_variable = { kr_intro_screen_variable = 4 }
				set_variable_to_random = kr_intro_screen_refresh
			}

			mod_options_button_click = {
				clear_variable = kr_intro_screen_variable
				set_variable_to_random = kr_intro_screen_refresh
			}
		}
	}

	kr_intro_screen_tab_1 = {
		context_type = player_context
		window_name = kr_intro_screen_tab_1_container
		parent_window_name = "kr_intro_screen_container"
		dirty = kr_intro_screen_refresh

		ai_enabled = { always = no }

		visible = {
			check_variable = { kr_intro_screen_variable = 1 }
		}

		triggers = {
			sound_events_button_visible = { has_kr_music_submod = yes }
			sound_events_text_visible = { has_kr_music_submod = yes }

			country_intro_page_indicator_box_visible = { has_variable = country_intro_page_count }
			country_intro_page_indicator_visible = { has_variable = country_intro_page_count }
			country_flip_back_visible = { has_variable = country_intro_page_count }
			country_flip_forward_visible = { has_variable = country_intro_page_count }

			country_flip_back_click_enabled = { check_variable = { curr_page_country > 0 } }
			country_flip_forward_click_enabled = { check_variable = { curr_page_country < country_intro_page_count } }
		}

		effects = {
			## Options
			custom_achievements_button_click = {
				if = {
					limit = { has_global_flag = disallow_achievements_flag }
					set_variable = { global.custom_achievements_frame = 2 }
					clr_global_flag = disallow_achievements_flag
				}
				else = {
					set_variable = { global.custom_achievements_frame = 1 }
					set_global_flag = disallow_achievements_flag
				}
				set_variable_to_random = kr_intro_screen_refresh
			}

			rename_button_click = {
				if = {
					limit = { has_global_flag = allowrenaming_flag }
					set_variable = { global.renaming_frame = 1 }
					clr_global_flag = allowrenaming_flag
				}
				else = {
					set_variable = { global.renaming_frame = 2 }
					set_global_flag = allowrenaming_flag
				}
				every_state = { rename_state_effect = yes }
				set_variable_to_random = kr_intro_screen_refresh
			}

			sound_events_button_click = {
				if = {
					limit = { has_global_flag = allow_sound_events_flag }
					set_variable = { global.sound_events_frame = 1 }
					clr_global_flag = allow_sound_events_flag
				}
				else = {
					set_variable = { global.sound_events_frame = 2 }
					set_global_flag = allow_sound_events_flag
				}
				set_variable_to_random = kr_intro_screen_refresh
			}

			regional_news_button_click = {
				if = {
					limit = { has_global_flag = NEWS_SOME }
					set_variable = { global.regional_news_frame = 1 }
					clr_global_flag = NEWS_SOME

					#also disable all news
					set_variable = { global.world_news_frame = 1 }
					clr_global_flag = NEWS_ALL
				}
				else = {
					set_variable = { global.regional_news_frame = 2 }
					set_global_flag = NEWS_SOME
				}
				set_variable_to_random = kr_intro_screen_refresh
			}

			world_news_button_click = {
				if = {
					limit = { has_global_flag = NEWS_ALL }
					set_variable = { global.world_news_frame = 1 }
					clr_global_flag = NEWS_ALL
				}
				else = {
					set_variable = { global.world_news_frame = 2 }
					set_global_flag = NEWS_ALL

					#also enable regional news
					set_variable = { global.regional_news_frame = 2 }
					set_global_flag = NEWS_SOME
				}
				set_variable_to_random = kr_intro_screen_refresh
			}

			## Country intro
			country_flip_back_click = {
				subtract_from_variable = { curr_page_country = 1 }
				set_variable_to_random = kr_intro_screen_refresh
			}
			country_flip_forward_click = {
				add_to_variable = { curr_page_country = 1 }
				set_variable_to_random = kr_intro_screen_refresh
			}
		}

		properties = {
			kr_intro_screen_tab_1_background_icon = {
				image = "[ROOT.GetCountryIntroBackground]"
			}

			## Options
			custom_achievements_button = {
				frame = global.custom_achievements_frame
			}
			rename_button = {
				frame = global.renaming_frame
			}
			sound_events_button = {
				frame = global.sound_events_frame
			}
			regional_news_button = {
				frame = global.regional_news_frame
			}
			world_news_button = {
				frame = global.world_news_frame
			}
		}
	}

	kr_intro_screen_tab_2 = {
		context_type = player_context
		window_name = kr_intro_screen_tab_2_container
		parent_window_name = "kr_intro_screen_container"
		dirty = kr_intro_screen_refresh

		ai_enabled = { always = no }

		visible = {
			check_variable = { kr_intro_screen_variable = 2 }
		}

		triggers = {
			kr_intro_screen_tab_2_reveal_spoilers_button_visible = { NOT = { has_variable = kr_intro_screen_spoilers_revealed } }
			kr_intro_screen_tab_2_header_visible = { has_variable = kr_intro_screen_spoilers_revealed }
			kr_intro_screen_tab_2_content_visible = { has_variable = kr_intro_screen_spoilers_revealed }

			country_guide_page_indicator_box_visible = {
				has_variable = kr_intro_screen_spoilers_revealed
				has_variable = country_guide_page_count
			}
			country_guide_page_indicator_visible = {
				has_variable = kr_intro_screen_spoilers_revealed
				has_variable = country_guide_page_count
			}
			guide_flip_back_visible = {
				has_variable = kr_intro_screen_spoilers_revealed
				has_variable = country_guide_page_count
			}
			guide_flip_forward_visible = {
				has_variable = kr_intro_screen_spoilers_revealed
				has_variable = country_guide_page_count
			}

			guide_flip_back_click_enabled = { check_variable = { curr_page_guide > 0 } }
			guide_flip_forward_click_enabled = { check_variable = { curr_page_guide < country_guide_page_count } }
		}

		effects = {
			kr_intro_screen_tab_2_reveal_spoilers_button_click = {
				set_variable = { kr_intro_screen_spoilers_revealed = 1 }
				set_variable = { curr_page_guide = 0 }
				set_variable_to_random = kr_intro_screen_refresh
			}

			guide_flip_back_click = {
				subtract_from_variable = { curr_page_guide = 1 }
				set_variable_to_random = kr_intro_screen_refresh
			}
			guide_flip_forward_click = {
				add_to_variable = { curr_page_guide = 1 }
				set_variable_to_random = kr_intro_screen_refresh
			}
		}
	}

	kr_intro_screen_tab_3 = {
		context_type = player_context
		window_name = kr_intro_screen_tab_3_container
		parent_window_name = "kr_intro_screen_container"
		dirty = kr_intro_screen_refresh

		ai_enabled = { always = no }

		visible = {
			check_variable = { kr_intro_screen_variable = 3 }
		}
	}

	kr_intro_screen_tab_4 = {
		context_type = player_context
		window_name = kr_intro_screen_tab_4_container
		parent_window_name = "kr_intro_screen_container"
		dirty = kr_intro_screen_refresh

		ai_enabled = { always = no }

		visible = {
			check_variable = { kr_intro_screen_variable = 4 }
		}
	}
}

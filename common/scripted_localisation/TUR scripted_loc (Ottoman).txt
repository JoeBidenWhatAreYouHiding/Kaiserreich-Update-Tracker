	name = OTT_GetPossibleCentralisationStates
	# Recursive, edit with caution
	text = {
		trigger = {
			check_variable = { OTT_current_centralisation_state_index < TUR.OTT_centralisation_states^num }
			set_temp_variable = { OTT_current_centralisation_state = TUR.OTT_centralisation_states^OTT_current_centralisation_state_index }
			add_to_temp_variable = { OTT_current_centralisation_state_index = 1 }
		}

	}
	text = {
		trigger = { check_variable = { TUR.OTT_centralisation_states^num = 0 } }
		localization_key = OTT_no_states_can_be_centralised
	}
}

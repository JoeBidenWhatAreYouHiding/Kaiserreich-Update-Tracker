faction_get_core_states_not_fully_controlled = {
	hidden_trigger = {
		tooltip_evaluation = eval
		set_temp_variable = { faction_core_states_not_fully_controlled = 0 }
		all_of_scopes = {
			array = faction_members
			all_owned_state = {
				if = {
					limit = {
						is_core_of = PREV
						NOT = { is_fully_controlled_by = PREV }
					}
					add_to_temp_variable = { faction_core_states_not_fully_controlled = 1 }
				}
			}
		}
	}
}

faction_get_enemy_core_states_controlled_by_us = {
	hidden_trigger = {
		tooltip_evaluation = eval
		set_temp_variable = { faction_enemy_core_states_controlled_by_us = 0 }
		all_of_scopes = {
			array = faction_members
			all_controlled_state = {
				if = {
					limit = {
						owner = { has_war_with = ROOT }
						is_core_of = owner
					}
					add_to_temp_variable = { faction_enemy_core_states_controlled_by_us = 1 }
				}
			}
		}
	}
}

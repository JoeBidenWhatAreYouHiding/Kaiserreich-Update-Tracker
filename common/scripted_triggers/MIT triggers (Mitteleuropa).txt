MIT_has_political_power_for_investment = {
	multiply_temp_variable = {
		MIT_value = {
			#MIT_cost_multiplier
			value = 1
			add = modifier@mitteleuropa_investment_cost
			clamp = { min = 0.1 max = 2 }
			#min = 0.1
		}
	}

	NOT = { check_variable = { political_power < MIT_value } }
	#NOT = { has_political_power < MIT_value } #has_political_power is inaccurate for small values atm
}

MIT_can_recover_political_power_investment = {
	multiply_temp_variable = {
		MIT_value = {
			#MIT_cost_multiplier
			value = 1
			add = modifier@mitteleuropa_investment_cost
			clamp = { min = 0.1 max = 2 }
			#min = 0.1
		}
	}
	NOT = { check_variable = { MIT_agenda_investment < MIT_value } }
}

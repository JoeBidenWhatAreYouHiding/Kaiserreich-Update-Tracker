can_unlock_second_track_of_sf_doctrine = {
	if = {
		limit = { check_variable = { sf_trees > 1 } }
		custom_trigger_tooltip = {
			tooltip = can_unlock_second_track_special_forces_doctrine_yes
			check_variable = { sf_trees > 1 }
		}
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = can_unlock_second_track_special_forces_doctrine_no
			always = no
		}
	}
}

# War Russian Roulette.
# Because we heard that KR is boring and too railroaded.
country_event = {
	id = april.1
	title = april.1.t
	desc = april.1.d
	picture = GFX_report_event_RUS_gun

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event april.1 option april.1.a"
		name = april.1.a
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				any_other_country = {
					is_ai = no
				}
			}
		}
		custom_effect_tooltip = one_random_war_every_year
		hidden_effect = {
			random_other_country = {
				limit = {
					is_ai = yes
				}
				random_other_country = {
					limit = {
						is_ai = yes
						can_declare_war_on = PREV
					}
					declare_war_on = {
						target = PREV
						type = annex_everything
					}
				}
			}
			country_event = { id = april.1 days = 365 }
			country_event = { id = april.2 days = 4 }
		}
	}

	option = {
		log = "[GetLogInfo]: event april.1 option april.1.b"
		name = april.1.b
		trigger = {
			is_ai = no
		}

		custom_effect_tooltip = will_crash_game
		hidden_effect = { crash_the_game = yes }
	}

	option = { # Bounce the event back to a player if they tag-switched and old tag receives event
		log = "[GetLogInfo]: event april.1 option april.1.c"
		name = april.1.c
		trigger = {
			is_ai = yes
		}
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				all_other_country = {
					is_ai = yes
				}
			}
		}

		random_other_country = {
			limit = {
				is_ai = no
			}
			country_event = april.1
		}
	}
}

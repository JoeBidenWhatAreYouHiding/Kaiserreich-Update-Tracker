### End of the Civil War ###
country_event = {
	id = american_civil_war_events.2400 #system event to end the civil war
	hidden = yes

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		exists = yes
		has_global_flag = USA_civil_war_begun_flag
		NOT = {
			has_war_with = CAN
			has_war_with = NEE
			has_country_flag = ACW_has_deal_with_devil
		}
		if = {
			limit = { MEX = { is_in_faction = no } }
			NOT = { has_war_with = MEX }
		}
		if = {
			limit = { NOT = { tag = WCA } }
			NOT = { country_exists = WCA }
		}
		if = {
			limit = { NOT = { tag = USA } }
			NOT = { country_exists = USA }
		}
		if = {
			limit = { NOT = { tag = APG } }
			NOT = { country_exists = APG }
		}
		if = {
			limit = { NOT = { tag = ACC } }
			OR = {
				NOT = { country_exists = ACC }
				is_in_faction_with = ACC
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event american_civil_war_events.2400 option 1"
		USA_civil_war_ends = yes
		USA_demobilize_forces = yes
		remove_ACW_claims = yes

		### Events for other tags
		LEC = { country_event = { id = legation_cities_events.150 days = 5 random_days = 10 } }
		HAI = { country_event = { id = haiti_events.21 days = 5 random_days = 5 } }

		if = {
			limit = {
				OR = {
					country_exists = USA
					country_exists = WCA
					country_exists = APG
					country_exists = ACC
					NEE = {
						exists = yes
						has_country_flag = NEE_entered_the_civil_war
					}
				}
			}
			every_country = {
				limit = {
					OR = {
						tag = USA
						tag = WCA
						tag = ACC
						tag = APG
						AND = {
							tag = NEE
							has_country_flag = NEE_entered_the_civil_war
						}
					}
				}
				country_event = american_civil_war_events.24
			}
		}
		else_if = {
			limit = { country_exists = NEE }
			NEE = { country_event = american_civil_war_events.24 }
		}
		else = {
			var:361.owner = { country_event = american_civil_war_events.24 }
		}
	}
}

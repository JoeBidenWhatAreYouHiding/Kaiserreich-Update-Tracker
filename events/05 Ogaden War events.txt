country_event = { ## Victory! (End war in Defence Phase)
	id = ogaden_war_events.3
	title = ogaden_war_events.3.t
	desc = {
		text = ogaden_war_events.3.desc.ETH
		trigger = { tag = ETH }
	}
	desc = {
		text = ogaden_war_events.3.desc.SOM
		trigger = { tag = SOM }
	}
	picture = GFX_report_event_MAF_askari2

	is_triggered_only = yes

	option = { ## End the War
		log = "[GetLogInfo]: event ogaden_war_events.3 option ogaden_war_events.3.a"
		name = ogaden_war_events.3.a
		if = {
			limit = { 906 = { is_claimed_by = ROOT } } #Ogaden
			remove_state_claim = 906 #Ogaden
		}
		event_target:ogaden_war_aggressor = { white_peace_with_ROOT = yes }
	}
	option = { ## Keep fighting
		log = "[GetLogInfo]: event ogaden_war_events.3 option ogaden_war_events.3.b"
		name = ogaden_war_events.3.b
		ai_chance = { base = 0 }
	}
	after = {
		hidden_effect = { event_target:ogaden_war_aggressor = { country_event = ogaden_war_events.6 } }
	}
}

country_event = { ## Defeat. (End war in Attack Phase)
	id = ogaden_war_events.6
	title = ogaden_war_events.6.t
	desc = {
		text = ogaden_war_events.6.desc.ETH
		trigger = {
			tag = ETH
			NOT = { has_war_with = FROM }
		}
	}
	desc = {
		text = ogaden_war_events.6.desc.SOM
		trigger = {
			tag = SOM
			NOT = { has_war_with = FROM }
		}
	}
	desc = {
		text = ogaden_war_events.6.desc.war
		trigger = { has_war_with = FROM }
	}
	picture = GFX_report_event_soldiers_marching

	is_triggered_only = yes

	immediate = {
		clear_global_event_target = ogaden_war_defender
		clear_global_event_target = ogaden_war_aggressor
	}

	option = { ## FROM ends the War
		log = "[GetLogInfo]: event ogaden_war_events.6 option ogaden_war_events.6.a"
		name = ogaden_war_events.6.a
		trigger = { NOT = { has_war_with = FROM } }
		effect_tooltip = { ROOT = { white_peace = FROM } }
	}
	option = { ## FROM keeps fighting
		log = "[GetLogInfo]: event ogaden_war_events.6 option ogaden_war_events.6.b"
		name = ogaden_war_events.6.b
		trigger = { has_war_with = FROM }
	}
}

country_event = { # loop event for unit desertions during ARG-CHL war (donotlog)
	id = argentina_chilean_war_events.100
	hidden = yes
	is_triggered_only = yes

	trigger = {
		ARG = { has_war_with = FOP }
	}

	immediate = {
		country_event = { id = argentina_chilean_war_events.100 days = 24 random_days = 10 random_hours = 24 }
	}

	option = {
		if = {
			limit = {
				ARG = {
					is_ai = yes
					surrender_progress > 0.4
					has_capitulated = no
				}
			}
			ARG = { country_event = argentina_chilean_war_events.101 }
		}
		else_if = {
			limit = {
				FOP = {
					is_ai = yes
					surrender_progress > 0.4
					has_capitulated = no
				}
			}
			FOP = { country_event = argentina_chilean_war_events.102 }
		}
		### Since it doesn't make sense for CHL to join ARG, instead reduce available manpower and inf equipment
		if = {
			limit = {
				CHL = {
					is_ai = yes
					has_war_with = ARG
					surrender_progress > 0.4
					has_capitulated = no
				}
			}
			CHL = { country_event = argentina_chilean_war_events.103 }

			### Die faster pls, thank you - solves CHL sitting in northern mountains with 100% war support
			if = {
				limit = {
					CHL = { surrender_progress > 0.7 }
				}
				CHL = {
					add_stability = -0.25
					add_war_support = -0.25
					disband_twenty_percent_units_delete_equipment = yes
				}
			}
		}
	}
}

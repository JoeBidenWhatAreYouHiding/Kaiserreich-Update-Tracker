country_event = { #Formal protest from countries with ITRO problem
	id = bulgaria_flavour_events.35
	title = bulgaria_flavour_events.35.t
	desc = bulgaria_flavour_events.35.d
	picture = GFX_report_event_BUL_assembly2

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			set_global_flag = BUL_ITRO_diplomatic_protest
			every_other_country = {
				limit = { NOT = { has_country_flag = BUL_protesting_itro } } #let the cancel_effect run instead, to recoup the decision cost
				remove_targeted_decision = { decision = BUL_ITRO_issue_formal_protest target = ROOT }
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event bulgaria_flavour_events.35 option bulgaria_flavour_events.35.a"
		name = bulgaria_flavour_events.35.a
		ai_chance = { base = 20 }
	}

	option = {
		log = "[GetLogInfo]: event bulgaria_flavour_events.35 option bulgaria_flavour_events.35.b"
		name = bulgaria_flavour_events.35.b
		trigger = {
			NOT = { has_government = national_populist }
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				FROM = {
					original_tag = GRE
					OR = {
						has_completed_focus = GRE_diplomacy_focus
						NOT = { is_ally_with = SER }
					}
				}
			}
		}
		set_country_flag = BUL_ITRO_formally_disbanded
		add_political_power = -25
		every_state = {
			limit = { has_dynamic_modifier = { modifier = BUL_ITRO_modifier } }
			BUL_decrease_ITRO_strength = yes
		}
	}

	option = {
		log = "[GetLogInfo]: event bulgaria_flavour_events.35 option bulgaria_flavour_events.35.c"
		name = bulgaria_flavour_events.35.c
		trigger = {
			has_government = national_populist
		}
		ai_chance = {
			base = 25
			modifier = {
				add = 25
				has_war = no
			}
		}
		set_country_flag = BUL_ITRO_strengthened
		add_war_support = 0.05
		every_state = {
			limit = { has_dynamic_modifier = { modifier = BUL_ITRO_modifier } }
			BUL_increase_ITRO_strength = yes
		}
	}
	after = {
		every_country = {
			limit = {
				any_of_scopes = {
					array = global.BUL_ITRO_states
					is_owned_by = PREV
				}
			}
			country_event = { id = bulgaria_flavour_events.36 days = 5 }
		}
	}
}

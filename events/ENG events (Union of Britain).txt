	trigger = {

				flag = ENG_tuc_federalist
				value > 1
			}
			has_country_flag = {
				flag = ENG_tuc_autonomist
				value > 1
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event union_of_britain_events.6 option union_of_britain_events.6.a"
		name = union_of_britain_events.6.a
		add_political_power = -50
		promote_character = ENG_central_committee
		hidden_effect = {
			if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_maximist
						value > 1
					}
				}
				country_event = { id = union_of_britain_events.7 days = 20 }
			}
			else_if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_federalist
						value > 1
					}
				}
				country_event = { id = union_of_britain_events.8 days = 20 }
			}
			else_if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_autonomist
						value > 1
					}
				}
				country_event = { id = union_of_britain_events.9 days = 20 }
			}
		}
	}

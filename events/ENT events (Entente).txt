country_event = {
	id = entente_events.2
	title = entente_events.2.t
	desc = entente_events.2.d
	picture = GFX_report_event_generic_diplomacy2

	is_triggered_only = yes

	trigger = {
		is_in_economic_sphere = no
		OR = {
			is_in_faction_with = FROM
			AND = {
				has_socialist_government = no
				at_war_with_or_justifying_wargoal_against_FROM = no
				NOT = { has_country_flag = CAN_refused_iedc }
				NOT = { has_country_flag = CAN_invited_to_iedc }
				OR = {
					is_subject = no
					AND = {
						is_subject = yes
						overlord = { is_in_same_economic_sphere_as_ROOT = yes }
					}
				}
				if = {
					limit = { is_in_faction = yes }
					NOT = { any_allied_country = { at_war_with_or_justifying_wargoal_against_FROM = yes } }
				}
				else_if = {
					limit = { num_subjects > 0 }
					NOT = { any_subject_country = { at_war_with_or_justifying_wargoal_against_FROM = yes } }
				}
			}
		}
	}

	immediate = {
		set_country_flag = CAN_invited_to_iedc

	option = {
		log = "[GetLogInfo]: event entente_events.2 option entente_events.2.a"
		name = entente_events.2.a #join +25PP
		ai_chance = {
			base = 50
			modifier = {
				factor = 0.8
				has_political_power > 50
			}

				factor = 0.5
				has_political_power > 100

				factor = 0.2

			FROM = { country_event = entente_events.3 }

	}
	option = {
		log = "[GetLogInfo]: event entente_events.2 option entente_events.2.b"
		name = entente_events.2.b #join +50PP
		ai_chance = {
			base = 30
			modifier = {
				factor = 0
				num_of_factories < 10
			}
			modifier = {
				factor = 0
				has_political_power < 50
			}
			modifier = {
				factor = 2
				has_political_power > 100
			}
			modifier = {
				factor = 2
				has_political_power > 150
			}
		}
		if = {
			limit = { GER_is_in_mitteleuropa = yes }
			GER_leave_mitteleuropa = yes
			custom_effect_tooltip = tooltip_white_line
		}
		custom_effect_tooltip = CAN_joins_iedc
		set_temp_variable = { ENT_pp_investment = 50 }
		CAN_add_iedc_pp = yes
		hidden_effect = {
			FROM = { country_event = entente_events.3 }
			country_event = { id = entente_events.5 days = 1 }
		}
	}
	option = {
		log = "[GetLogInfo]: event entente_events.2 option entente_events.2.c"
		name = entente_events.2.c #join +100PP
		ai_chance = {
			base = 20
			modifier = {

# IEDC shut down
country_event = {
	id = entente_events.13
	title = entente_events.13.t
	desc = entente_events.13.d
	picture = GFX_report_event_CAN_commission

	fire_only_once = yes

	trigger = {
		tag = event_target:iedc_leader
		is_economic_sphere_leader = yes
		check_variable = { economic_sphere:members^num < 2 }
		has_global_flag = { flag = CAN_iedc_created days > 180 }
	}

	mean_time_to_happen = {
		days = 7
	}

	option = {
		log = "[GetLogInfo]: event entente_events.13 option entente_events.13.a"
		name = entente_events.13.a
		clr_global_flag = CAN_iedc_created
		if = {
			limit = { has_idea = CAN_iedc_main }
			remove_ideas = CAN_iedc_main
		}
		CAN_remove_from_iedc_effect = yes
		hidden_effect = {
			every_country = {
				clr_country_flag = CAN_refused_iedc
				clr_country_flag = CAN_invited_to_iedc
				CAN_remove_from_iedc = yes
			}
			clear_variable = global.iedc_pp
			dismantle_economic_sphere = yes
			clear_global_event_target = iedc_leader
		}
	}
}

	title = entente_events.355.t
	desc = entente_events.355.d
	picture = GFX_report_event_CAN_toronto

	trigger = {
		has_character = GBR_albert_windsor
		is_monarchy = yes
	}

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event entente_events.355 option entente_events.355.a"
		name = entente_events.355.a
		add_political_power = -30
		if = {
			limit = { check_variable = { king_influence > 0 } }
			custom_effect_tooltip = CAN_exile_influence_decrease
			subtract_from_variable = { king_influence = 1 }
			CAN_calc_exile_influence_effects = yes
		}
	}
}

country_event = { #Albert improves
	id = entente_events.356
	title = entente_events.356.t
	desc = entente_events.356.d
	picture = GFX_report_event_CAN_microphone

	trigger = {
		has_country_leader = { character = GBR_albert_windsor }
	}

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event entente_events.356 option entente_events.356.a"
		name = entente_events.356.a
		add_political_power = 30
		effect_tooltip = {
			swap_ruler_traits = {
				remove = GBR_humble
				add = GBR_humble_improved
			}
		}
		hidden_effect = {
			var:country_leader = {
				set_temp_variable = { remove_trait_token = token:GBR_humble }
				set_temp_variable = { add_trait_token = token:GBR_humble_improved }
				swap_traits_in_all_country_leader_roles = yes
			}
		}
	}
}

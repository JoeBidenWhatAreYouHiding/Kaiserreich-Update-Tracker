country_event = { #Assassination attempt on Sukarno
	id = indonesia_events.1
	title = indonesia_events.1.t
	desc = indonesia_events.1.d
	picture = GFX_report_event_INS_independence_rally

	fire_only_once = yes

	trigger = {
		tag = INS
		has_government = authoritarian_democrat
		OR = {
			focus_progress = { focus = INS_political_crossroad progress > 0.5 }
			has_completed_focus = INS_political_crossroad
		}
		OR = {
			has_country_flag = ins_medium_unity
			has_country_flag = ins_low_unity
			AND = {
				is_ai = yes
				OR = {
					has_game_rule = { rule = INS_path option = INS_PATH_ISLAMIST_PATAUT }
					has_game_rule = { rule = INS_path option = INS_PATH_SULTANATE_PATAUT }
				}
			}
		}
		if = {
			limit = {
				is_ai = yes
				NOT = {
					has_game_rule = { rule = INS_path option = INS_PATH_ISLAMIST_PATAUT }
					has_game_rule = { rule = INS_path option = INS_PATH_SULTANATE_PATAUT }
				}
			}
			NOT = {
				has_game_rule = { rule = INS_path option = INS_PATH_SOCCON }
				has_game_rule = { rule = INS_path option = INS_PATH_AUTDEM }
				has_game_rule = { rule = INS_path option = INS_PATH_RADSOC }
			}
		}
	}

	immediate = {
		hidden_effect = {
			country_event = { id = indonesia_events.2 days = 20 }
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = { #Sukarno survives
		log = "[GetLogInfo]: event indonesia_events.1 option indonesia_events.1.a"
		name = indonesia_events.1.a
		ai_chance = { base = 20 }
	}

	option = { #Sukarno dies
		log = "[GetLogInfo]: event indonesia_events.1 option indonesia_events.1.b"
		name = indonesia_events.1.b
		ai_chance = { base = 80 }
		retire_character = INS_sukarno
		add_country_leader_role = {
			character = INS_mohammad_hatta
			country_leader = {
				desc = INS_mohammad_hatta_leader_desc
				ideology = authoritarian_democrat_subtype
			}
			promote_leader = yes
		}
	}

	after = {
		add_stability = -0.1
	}
}

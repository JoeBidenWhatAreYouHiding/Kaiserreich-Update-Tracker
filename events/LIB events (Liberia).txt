	id = liberia_events.57
	title = liberia_events.57.t
	desc = liberia_events.57.d
	picture = GFX_report_event_USA_congress

	is_triggered_only = yes

	option = { #Fine, if they want it so badly they can have it
		log = "[GetLogInfo]: event liberia_events.57 option liberia_events.57.a"
		name = liberia_events.57.a
		ai_chance = {
			base = 25
			modifier = {
				factor = 0
				tag = IVO
			}
			modifier = {
				factor = 3
				strength_ratio = {
					tag = FROM
					ratio < 1.0
				}
			}
			modifier = {
				factor = 0.5
				strength_ratio = {
					tag = FROM
					ratio > 2.0
				}
			}
			modifier = {
				factor = 0
				strength_ratio = {
					tag = FROM
					ratio > 3.0
				}
			}
			modifier = {
				factor = 2
				has_war = yes
			}
			modifier = {
				factor = 2
				surrender_progress > 0.20
			}
		}
		if = {
			limit = { tag = IVO }
			custom_effect_tooltip = GAME_OVER_TT
			effect_tooltip = {
				FROM = { annex_country = { target = ROOT } }
			}
		}
		else = {
			add_opinion_modifier = {
				target = FROM
				modifier = KR_outraged
			}
			add_political_power = -50
			effect_tooltip = {
				if = {
					limit = { has_subject = IVO }
					FROM = { annex_country = { target = IVO } }
				}
				else_if = {
					limit = { 807 = { is_owned_by_ROOT_or_subject = yes } }
					FROM = { transfer_state = 807 }
				}
			}
		}
		FROM = { country_event = { id = liberia_events.59 hours = 8 } }
	}
	option = { #How dare they! We will push them back across the border and beyond!
		log = "[GetLogInfo]: event liberia_events.57 option liberia_events.57.b"
		name = liberia_events.57.b
		ai_chance = { base = 75 }
		FROM = { country_event = { id = liberia_events.58 hours = 8 } }
	}
}

# Peacing out with ACW victor
country_event = {
	id = liberia_events.93
	title = liberia_events.93.t
	desc = liberia_events.93.d
	picture = GFX_report_event_generic_diplomacy2

	fire_only_once = yes

	trigger = {
		tag = LIB
		NOT = { country_exists = USA }
		is_subject = no
		is_in_faction = no
		has_war = yes
		OR = {
			WCA = {
				has_war_with = ROOT
				NOT = { is_neighbor_of = ROOT }
				NOT = { has_war_with_wargoal_against = ROOT }
				NOT = { ROOT = { has_war_with_wargoal_against = ROOT } }
				is_valid_peace_target = yes
			}
			APG = {
				has_war_with = ROOT
				NOT = { is_neighbor_of = ROOT }
				NOT = { has_war_with_wargoal_against = ROOT }
				NOT = { ROOT = { has_war_with_wargoal_against = ROOT } }
				is_valid_peace_target = yes
			}
			ACC = {
				has_war_with = ROOT
				NOT = { is_neighbor_of = ROOT }
				NOT = { has_war_with_wargoal_against = ROOT }
				NOT = { ROOT = { has_war_with_wargoal_against = ROOT } }
				is_valid_peace_target = yes
			}
			NEE = {
				has_war_with = ROOT
				NOT = { is_neighbor_of = ROOT }
				NOT = { has_war_with_wargoal_against = ROOT }
				NOT = { ROOT = { has_war_with_wargoal_against = ROOT } }
				is_valid_peace_target = yes
			}
		}
	}

	mean_time_to_happen = {
		days = 3
	}

	option = { #We will go it alone
		log = "[GetLogInfo]: event liberia_events.93 option liberia_events.93.a"
		name = liberia_events.93.a
		ai_chance = { base = 25 }
		every_enemy_country = {
			limit = {
				is_valid_peace_target = yes
				OR = {
					is_american_tag = yes
					tag = ENT
					tag = CAN
					original_tag = MEX
				}
			}
			white_peace_with_ROOT = yes
		}
		add_war_support = 0.15
	}

	option = {
		log = "[GetLogInfo]: event liberia_events.93 option liberia_events.93.b"
		name = liberia_events.93.b #Germany
		trigger = {
			has_socialist_government = no
			country_exists = GER
			NOT = { has_war_with = GER }
			GER = {
				has_socialist_government = no
				is_subject = no
				has_capitulated = no
				NOT = { has_country_flag = LIB_refused_bailout }
			}
		}
		ai_chance = { base = 25 }
		every_enemy_country = {
			limit = {
				is_valid_peace_target = yes
				OR = {
					is_american_tag = yes
					tag = ENT
					tag = CAN
					original_tag = MEX
				}
			}
			white_peace_with_ROOT = yes
		}
		GER = { country_event = liberia_events.87 }
	}

	option = {
		log = "[GetLogInfo]: event liberia_events.93 option liberia_events.88.c"
		name = liberia_events.88.c #Firestone
		ai_chance = { base = 20 }
		every_enemy_country = {
			limit = {
				is_valid_peace_target = yes
				OR = {
					is_american_tag = yes
					tag = ENT
					tag = CAN
					original_tag = MEX
				}
			}
			white_peace_with_ROOT = yes
		}
		USA = { clr_country_flag = LIB_refused_bailout }
		GER = { clr_country_flag = LIB_refused_bailout }
		add_ideas = LIB_firestone_bailout_idea
		add_popularity = {
			ideology = market_liberal
			popularity = 0.1
		}
	}

	after = {
		if = {
			limit = {
				has_idea = LIB_bailout_usa_idea
			}
			remove_ideas = LIB_bailout_usa_idea
		}
		if = {
			limit = {
				has_idea = LIB_bailout_usa_2_idea
			}
			remove_ideas = LIB_bailout_usa_2_idea
		}
		if = {
			limit = {
				has_idea = LIB_tax_exemption_idea
			}
			remove_ideas = LIB_tax_exemption_idea
		}
	}

add_namespace = malta_events

# Ask to join Italy
country_event = {
	id = malta_events.1
	title = malta_events.1.t
	desc = malta_events.1.d
	picture = GFX_report_event_spain_civil_war_soldiers

	trigger = {
		tag = MLT
		has_war = no
		var:115.owner = {
			is_italian_tag = yes
			NOT = { has_war_with = ROOT }
			if = {
				limit = { ROOT = { is_subject = yes } }
				OR = {
					is_subject_of = ROOT.overlord
					AND = {
						is_in_faction_with = ROOT.overlord
						var:ROOT.overlord = { is_faction_leader = yes }
					}
				}
			}
			else = {
				is_subject = no
			}
			has_capitulated = no
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		var:115.owner = { save_event_target_as = italy }
	}

	option = {
		log = "[GetLogInfo]: event malta_events.1 option malta_events.1.a"
		name = malta_events.1.a # Ask for a Union
		ai_chance = { base = 60 }
		trigger = { is_subject = no }
		event_target:italy = { country_event = { id = malta_events.2 hours = 8 } }
	}

	option = {
		log = "[GetLogInfo]: event malta_events.1 option malta_events.1.b"
		name = malta_events.1.b # Ask overlord to allow Union
		ai_chance = { base = 60 }
		trigger = { is_subject = yes }
		overlord = { country_event = { id = malta_events.3 hours = 8 } }
	}

	option = {
		log = "[GetLogInfo]: event malta_events.1 option malta_events.1.c"
		name = malta_events.1.c # Don't ask
		ai_chance = {
			base = 40
			modifier = {
				factor = 0
				var:115.owner = { is_ai = no }
			}
			modifier = {
				factor = 0
				is_subject = yes
				overlord = { is_ai = no }
			}
		}
	}
}

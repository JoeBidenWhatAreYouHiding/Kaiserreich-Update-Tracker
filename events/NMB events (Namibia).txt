	desc = namibia_events.37.d
	is_triggered_only = yes
	fire_only_once = yes
	picture = GFX_report_event_african_soldiers

	option = {
		log = "[GetLogInfo]: event namibia_events.37 option namibia_events.37.a"
		name = namibia_events.37.a
		ai_chance = { base = 70 }
		save_event_target_as = NMB_send_supplies_generic
		add_equipment_to_stockpile = {
			type = infantry_equipment_1
			amount = -2000
		}
	}

	option = {
		log = "[GetLogInfo]: event namibia_events.37 option namibia_events.37.b"
		name = namibia_events.37.b
		ai_chance = { base = 30 }
	}

	after = {
		FROM = {
			country_event = { id = namibia_events.38 days = 1 }
		}
	}
}

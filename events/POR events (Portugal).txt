# The Assassination of the King of Portugal! \ Happens if the CNT-FAI Win, Reference To An Earlier Assassination

country_event = {
	id = portugal_events.304
	title = portugal_events.304.t
	desc = portugal_events.304.d
	picture = GFX_report_event_POR_assassination

	fire_only_once = yes

	trigger = {
		tag = POR
		is_neighbor_of = SWF
		SWF = {
			has_war = no
			is_subject = no
		}
		has_country_leader = { character = POR_duarte_braganca }
		is_subject = no
		has_war = no
		NOT = {
			has_global_flag = spanish_civil_war
		}
	}

	mean_time_to_happen = {
		days = 60
	}

	option = { # The CNT-FAI must surrender those responsible! / CNT-FAI Is Given An Ultimatum, With War As A Consequence
		log = "[GetLogInfo]: event portugal_events.304 option portugal_events.304.a"
		name = portugal_events.304.a
		ai_chance = { base = 90 }
		SWF = {
			country_event = { id = portugal_events.305 days = 10 }
		}
	}

	option = { # We must be cautious in this situation...
		log = "[GetLogInfo]: event portugal_events.304 option portugal_events.304.b"
		name = portugal_events.304.b
		ai_chance = {
			base = 10
			modifier = {
				factor = 9
				strength_ratio = {
					tag = SWF
					ratio < 0.5
				}
			}
		}
		add_stability = -0.05
		add_political_power = -50
		reverse_add_opinion_modifier = {
			target = SWF
			modifier = KR_outraged
		}
	}

	after = {
		POR_duarte_braganca = { kill_character = yes }
		POR_create_maria_adelaide = yes
	}
}
